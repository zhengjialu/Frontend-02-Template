> 用 JavaScript 去设计狗咬人的代码

```javascript
// 人
class Human {
  constructor(health) {
    // 是否健康
    this.isHealth = health
  }

  hurt(health) {
    console.log('被狗咬了...不健康了', !this.isHealth)
  }
}

// 狗
class Dog {
  constructor(bite) {
    // 是否咬人了
    this.isBite = false
  }

  bite(attack) {
    this.isBite = attack
    console.log('咬那个憨憨', this.isBite)
  }
}

// ----------------------------------------------

let dogInstance = new Dog(false)
dogInstance.bite(true)

let humanInstance = new Human(true)
humanInstance.hurt(dogInstance.isBite)
```
